------------------------------------------------------------------------------------------------------------------------------------------------------------
PREFIX dbo: <http://dbpedia.org/ontology/>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

WHERE {
	?politician rdf:type dbo:Politician.
    	?politician ?property ?value.
	FILTER (?property != rdf:type)
}
